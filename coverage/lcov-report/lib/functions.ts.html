/**
 * Wyodrbnia domen gwn z podanego adresu URL, usuwajc protok, subdomeny, cieki, parametry zapytania i fragmenty.
 *
 * @function extractDomain
 * @param {string} url - Adres URL do przetworzenia (np. "https://www.google.pl/path?query=1").
 * @returns {string} Znormalizowana domena gwna (np. "google.pl").
 *
 * @example
 * ```typescript
 * import { extractDomain } from '@/lib/functions';
 *
 * console.log(extractDomain('https://www.google.pl/path?query=1')); // "google.pl"
 * console.log(extractDomain('http://sub.example.com')); // "example.com"
 * console.log(extractDomain('invalid-url')); // "invalid-url"
 * ```
 *
 * @remarks
 * - Funkcja usuwa protok (`http://`, `https://`), subdomeny (`www.`, `sub.`), cieki (`/path`), parametry zapytania (`?query=1`) oraz fragmenty (`#fragment`).
 * - Jeli URL ma wicej ni dwie czci oddzielone kropkami (np. `sub.example.com`), zwracane s tylko ostatnie dwie czci (`example.com`).
 * - W przypadku niepoprawnego URL-a funkcja zwraca wejciowy cig bez zmian.
 */
export function extractDomain(url: string): string {
  try {
 
    let domain = url.replace(/^(https?:\/\/)?/i, '');
 
 
    domain = domain.split('/')[0];
 
 
    domain = domain.split('?')[0].split('#')[0];
 
 
    domain = domain.replace(/^www\./i, '');
 
    return domain && /^[a-z0-9.-]+$/.test(domain) ? domain : url;
  } catch {
    return url; 
  }
}

 * @returns {number} rednia sia hase w procentach (0-100), zaokrglona do najbliszej liczby cakowitej.
 *
 * @example
 * ```typescript
 * import { calculateAverageStrength } from '@/functions';

 * const history = [{ platform: "test", login: "user", strength: 80 }];

 * const strength = calculateAverageStrength(history, passwords); // Wynik: 80

 * - Funkcja filtruje histori, uwzgldniajc tylko rekordy powizane z istniejcymi hasami w `passwords`.
 * - rednia jest obliczana na podstawie pola `strength` w filtrowanej historii.
 *
 * @see {@link PasswordHistory} - Definicja typu `PasswordHistory`.
 * @see {@link PasswordTable} - Definicja typu `PasswordTable`.
 */
export const calculateAverageStrength = (history: PasswordHistory[], passwords : PasswordTable[]): number => {
  if (history.length === 0) return 100.0;
  const hist = [...history].filter((item) =>
    passwords.some(
      (p) => p.platform === item.platform && p.login === item.login
    )
  );
  const totalStrength = hist.reduce((sum, item) => sum + item.strength, 0);
  return Math.round(totalStrength / hist.length) || 100.0;
};

/**
 * Sprawdza, czy konto powizane z podanym adresem email jest zablokowane z powodu zbyt wielu nieudanych prb logowania.
 *
 * @function isEmailLockedOut
 * @param {string} email - Adres email do sprawdzenia.
 * @returns {boolean} `true`, jeli konto jest zablokowane, `false` w przeciwnym razie.
 *
 * @example
 * ```typescript
 * import { isEmailLockedOut, recordFailedAttempt } from '@/lib/functions';
 *
 * for (let i = 0; i < 5; i++) recordFailedAttempt("user@example.com");
 * console.log(isEmailLockedOut("user@example.com")); // true
 * ```
 *
 * @remarks
 * - Konto jest uznawane za zablokowane, jeli liczba nieudanych prb (`count`) osigna lub przekroczya `MAX_ATTEMPTS` i czas od ostatniej prby (`lastAttempt`) jest krtszy ni `LOCKOUT_DURATION`.
 * - Funkcja korzysta z danych zapisanych w `localStorage` przez `getFailedLogins`.
 *
 * @see {@link MAX_ATTEMPTS} - Stae `MAX_ATTEMPTS`.
 * @see {@link getFailedLogins} - Funkcja `getFailedLogins`.
 * @see {@link FailedLoginAttempt} - Definicja interfejsu `FailedLoginAttempt`.
 */
 export const isEmailLockedOut = (email: string): boolean => {
   const failedLogins = getFailedLogins();
   const attempt = failedLogins[email];
   if (!attempt || attempt.count < MAX_ATTEMPTS) return false;
 
   const timeSinceLastAttempt = Date.now() - attempt.lastAttempt;
   return timeSinceLastAttempt < LOCKOUT_DURATION;
 };
 
/**
 * Zwraca pozostay czas blokady konta w sekundach dla podanego adresu email.
 *
 * @function getRemainingLockoutTime
 * @param {string} email - Adres email do sprawdzenia.
 * @returns {number} Pozostay czas blokady w sekundach (0, jeli konto nie jest zablokowane).
 *
 * @example
 * ```typescript
 * import { getRemainingLockoutTime, recordFailedAttempt } from '@/lib/functions';
 *
 * for (let i = 0; i < 5; i++) recordFailedAttempt("user@example.com");
 * console.log(getRemainingLockoutTime("user@example.com")); // np. 599 (jeli blokada trwa 10 minut)
 * ```
 *
 * @remarks
 * - Funkcja zwraca 0, jeli konto nie jest zablokowane lub liczba nieudanych prb jest mniejsza ni `MAX_ATTEMPTS`.
 * - Pozostay czas jest obliczany jako rnica midzy `LOCKOUT_DURATION` a czasem od ostatniej nieudanej prby.
 * - Wynik jest zaokrglany w gr do najbliszej sekundy.
 *
 * @see {@link LOCKOUT_DURATION} - Staa `LOCKOUT_DURATION`.
 * @see {@link getFailedLogins} - Funkcja `getFailedLogins`.
 * @see {@link FailedLoginAttempt} - Definicja interfejsu `FailedLoginAttempt`.
 */
 export const getRemainingLockoutTime = (email: string): number => {
   const failedLogins = getFailedLogins();
   const attempt = failedLogins[email];
   if (!attempt || attempt.count < MAX_ATTEMPTS) return 0;
 
   const timeSinceLastAttempt = Date.now() - attempt.lastAttempt;
   const remainingTime = LOCKOUT_DURATION - timeSinceLastAttempt;
   return remainingTime > 0 ? Math.ceil(remainingTime / 1000) : 0;
 };
 
/**
 * Rejestruje nieudan prb logowania dla podanego adresu email.

 * @function recordFailedAttempt

 * ```typescript

 * recordFailedAttempt("user@example.com");

 * ```
 *
 * @remarks
 * - Funkcja zwiksza licznik nieudanych prb (`count`) i aktualizuje czas ostatniej prby (`lastAttempt`) dla podanego emaila.
 * - Dane s zapisywane w `localStorage` za pomoc funkcji `saveFailedLogins`.
 * - Jeli email nie istnieje w rekordach, tworzony jest nowy wpis z `count: 1` i biecym timestampem.

 * @see {@link getFailedLogins} - Funkcja `getFailedLogins`.
 * @see {@link saveFailedLogins} - Funkcja `saveFailedLogins`.
 * @see {@link FailedLoginAttempt} - Definicja interfejsu `FailedLoginAttempt`.
 */
 export const recordFailedAttempt = (email: string) => {
   const failedLogins = getFailedLogins();
   const attempt = failedLogins[email] || { count: 0, lastAttempt: 0 };
   attempt.count += 1;
   attempt.lastAttempt = Date.now();

 };
